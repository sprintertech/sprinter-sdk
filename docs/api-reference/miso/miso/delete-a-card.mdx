```python Python
import requests

response = requests.delete(
    "https://api.getmiso.fi/miso/cards/{cardId}",
    json={}
)
print(response.json())
```

```go Go
package main

import (
	"bytes"
	"fmt"
	"io"
	"net/http"
)

func main() {
	resp, _ := http.NewRequest("DELETE", "https://api.getmiso.fi/miso/cards/{cardId}", bytes.NewBuffer([]byte("{}")))
	resp.Header.Set("Content-Type", "application/json")
	client := &http.Client{}
	res, _ := client.Do(resp)
	defer res.Body.Close()
	body, _ := io.ReadAll(res.Body)
	fmt.Println(string(body))
}
```
