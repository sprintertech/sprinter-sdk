```go Go
package main

import (
	"bytes"
	"fmt"
	"io"
	"net/http"
)

func main() {
	resp, _ := http.NewRequest("POST", "https://api.getmiso.fi/miso/referral/validate", bytes.NewBuffer([]byte("{}")))
	resp.Header.Set("Content-Type", "application/json")
	client := &http.Client{}
	res, _ := client.Do(resp)
	defer res.Body.Close()
	body, _ := io.ReadAll(res.Body)
	fmt.Println(string(body))
}
```
